// Smart Contract for Voting
pragma solidity ^0.8.0;

contract Voting {
    struct Candidate {
        string name;
        uint voteCount;
    }
    
    Candidate[] public candidates;
    mapping(address => bool) public voters;
    
    constructor(string[] memory candidateNames) {
        for(uint i = 0; i < candidateNames.length; i++) {
            candidates.push(Candidate({
                name: candidateNames[i],
                voteCount: 0
            }));
        }
    }
    
    function vote(uint candidate) public {
        require(!voters[msg.sender], "Already voted");
        require(candidate < candidates.length, "Invalid candidate");
        
        voters[msg.sender] = true;
        candidates[candidate].voteCount++;
    }
}

// Activity tracking component
import React, { useState, useEffect } from 'react';
import { View, Text } from 'react-native';
import { Accelerometer } from 'expo-sensors';

const ActivityTracker = () => {
    const [steps, setSteps] = useState(0);
    const [subscription, setSubscription] = useState(null);

    useEffect(() => {
        Accelerometer.setUpdateInterval(1000);
        const sub = Accelerometer.addListener(accelerometerData => {
            // Step counting logic
            if (Math.abs(accelerometerData.x) > 1.5) {
                setSteps(prev => prev + 1);
            }
        });
        setSubscription(sub);

        return () => {
            subscription && subscription.remove();
        };
    }, []);

    return (
        
            Steps: {steps}
        
    );
};

export default ActivityTracker;

# Network packet analyzer
from scapy.all import *
import time

def packet_callback(packet):
    if packet.haslayer(IP):
        src_ip = packet[IP].src
        dst_ip = packet[IP].dst
        protocol = packet[IP].proto
        
        # Analyze packet for suspicious patterns
        if packet.haslayer(TCP):
            if packet[TCP].flags == 0x02:  # SYN scan detection
                print(f"Potential SYN scan detected from {src_ip}")
        
        # Log packet information
        log_packet(src_ip, dst_ip, protocol)

def log_packet(src_ip, dst_ip, protocol):
    timestamp = time.strftime("%Y-%m-%d %H:%M:%S")
    log_entry = f"{timestamp} | {src_ip} -> {dst_ip} | Protocol: {protocol}\n"
    
    with open("network_log.txt", "a") as f:
        f.write(log_entry)

# Start packet capture
sniff(prn=packet_callback, store=0)
